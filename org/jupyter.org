import mmnpz
from einops import rearrange
import numpy as np
import os.path as osp

# Load and access datasets
path_RAFT = "/fred/oz141/ldezoete/MS-G3D/data/ntu/aligned_data/poseoff/RAFT/ntu_CS-poseoff_RAFT_D1_aligned.npz"
path_LK = "/fred/oz141/ldezoete/MS-G3D/data/ntu/aligned_data/poseoff/LK/ntu_CS-poseoff_LK_D1_aligned.npz"
dat_RAFT = mmnpz.load(path_RAFT)
dat_LK = mmnpz.load(path_LK)

for file_name, file_data in {"RAFT":dat_RAFT, "LK":dat_LK}.items():
    print(file_name)
    batch_size=64

    org_data = file_data
    splits = ["x_train", "x_test"]

    count = None
    mean = None
    M2 = None

    # Original splits are arrays of shape (N, T, M*V*C)
    M, V = 2, 25
    C = 53 if "poseoff" in path_RAFT else 3

    # Process the splits :)
    for split in splits:
        data = org_data[split]
        N, T, _ = data.shape
        # Batch the data
        for start in range(0, N, batch_size):
            # Load and reshape batch
            batch = data[start:start+batch_size].astype(np.float64)
            batch = batch.reshape(batch.shape[0], T, M, V, C)
            # batch shape: (B, T, M, V, C)

            # A frame is "real" if any of (M, V, C) is non-zero
            valid_mask = (batch != 0).any(axis=(3,4)) # (B, T, M) - True if a frame is real!
            batch_count = valid_mask.sum()

            if batch_count == 0:
                continue

            # Extract mean and var only for valid frames
            valid_frames = batch[valid_mask]
            batch_mean = valid_frames.mean(axis=0) # V, C
            batch_var = valid_frames.var(axis=0) # V, C

            if count is None:
                count = batch_count
                mean = batch_mean
                M2 = batch_var * batch_count
            else:
                new_count = count + batch_count
                delta = batch_mean - mean
                mean = mean + delta * (batch_count / new_count)
                M2 = M2 + batch_var * batch_count + (delta ** 2) * (count*batch_count/new_count)
                count = new_count
            break
    # Calculate standard deviation after mean calculation
    variance = M2 / count
    std = np.sqrt(variance)
    # Reshape (V, C) -> (C, 1, V, 1)
    mean_map = mean.T[:, np.newaxis, :, np.newaxis]
    std_map = std.T[:, np.newaxis, :, np.newaxis]

    print(f"Mean map shape: {mean_map.shape}")
    print(f"Mean map zero {mean_map[:, 0, 0, 0]}")
    print(f"Standard deviation map shape: {std_map.shape}")
    print(f"Standard deviation map 0 {std_map[:, 0, 0, 0]}")
